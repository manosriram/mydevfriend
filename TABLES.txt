CREATE TABLE user (
    userid SERIAL,
    fullname VARCHAR(32),
    username VARCHAR(32),
    email VARCHAR(32),
    password VARCHAR(64),
    location VARCHAR(32),
    dob DATE,
    profilePic VARCHAR(64),
    created TIMESTAMP,
    primary key (username)
);

CREATE TABLE user_language (
    username varchar(32),
    language varchar(32),
    primary key (username, language)
);
